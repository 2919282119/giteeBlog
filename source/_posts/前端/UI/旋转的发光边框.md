## 代码
###
```vue
<script setup>
import { computed } from 'vue';

const props=defineProps(['width','height','radius','bandwidth'])
const width=computed(()=>props.width??"100vw")
const height=computed(()=>props.height??"90vh")
const radius=computed(()=>props.radius??"0px")
const bandwidth=computed(()=>props.bandwidth??"3vh")

</script>

<template>
    <div class="box">
        <div class="content">
            <slot></slot>
        </div>
    </div>
</template>

<style scoped lang="less">
@width: v-bind(width);
@height: v-bind(height);
@radius: v-bind(radius);
@bandwidth: v-bind(bandwidth);

@property --a {
    syntax: '<angle>';
    inherits: false;
    initial-value: 0deg;
}

@keyframes rotate {
    0% {
        --a: 0deg;
    }

    100% {
        --a: 360deg;
    }
}

.box {
    width: @width;
    height: @height;
    position: relative;
    background-image: repeating-conic-gradient(from var(--a),
            #ff2770 0%,
            #ff2770 5%,
            transparent 5%,
            transparent 40%,
            #ff2770 50%);
    animation: rotate 5s linear infinite;
    border-radius: @radius;
    display: flex;
    justify-content: center;
    align-items: center;
    z-index: 10;
    padding: @bandwidth;
}

.box::before {
    content: "";
    position: absolute;
    width: @width;
    height: @height;
    background-image: repeating-conic-gradient(from var(--a),
            #45f3ff 0%,
            #45f3ff 5%,
            transparent 5%,
            transparent 40%,
            #45f3ff 50%);
    animation: rotate 5s linear infinite;
    animation-delay: -1s;
    border-radius: @radius;
    z-index: -1;
}

.content {
    width:100%;
    height:100%;
    background-color: #fff;
    border-radius: 5px;
    overflow: hidden;
}
</style>
<style>
* {
    margin: 0;
    padding: 0;
    box-sizing: border-box;
}
</style>
```
